    var o = {};
    var a = this.serializeArray();

    $.each(a, function() {
        if (o[this.name]) {
            if (!o[this.name].push) {
                o[this.name] = [o[this.name]];
            }
            o[this.name].push(this.value || '');
        } else {
            o[this.name] = this.value || '';
        }
    });

    function isNoticeDate(endDate) {
        var cDate  = new Date();
        var cDay   = cDate.getDate();
        var cYear  = cDate.getFullYear();
        var cMonth = cDate.getMonth() + 1;

        if(parseInt(cMonth) < 10) {
            cMonth = "0" + cMonth;
        }

        if(parseInt(cDay) < 10) {
            cDay = "0" + cDay;
        }

        var curDate = cYear + "" + cMonth + "" + cDay;
        if(curDate <= endDate) {
            return true;
        } else {
            return false;
        }
    }

        for (var i=0;i<allCookies.length;i++){
            cookieArray=allCookies[i].split('=');
            if ("SEARCHKEYWORD"==cookieArray[0]) {
                initFlag = 'true';
                break;
            }
        }

        if(initFlag == 'true') {
            allCookies=document.cookie.split('; ');
            for (i=0;i<allCookies.length;i++){
                cookieArray=allCookies[i].split('=');
                if ("SEARCHKEYWORD"==cookieArray[0]) {
                    keywordArr = cookieArray[1].split('$');
                    if (keywordArr.length == 10) {
                        cookieArray[1] = "";
                        for(var j=0; j < keywordArr.length; j++) {
                            if(j == 9) {
                                cookieArray[1] += keyword;
                                break;
                            }
                            cookieArray[1] += (keywordArr[j+1] + "$");
                        }
                        document.cookie = "SEARCHKEYWORD" + "=" + cookieArray[1] + "; path=/; expires=" + todayDate.toGMTString();
                    } else {
                        cookieArray[1] += "$" + keyword;
                        document.cookie = "SEARCHKEYWORD" + "=" + cookieArray[1] + "; path=/; expires=" + todayDate.toGMTString();
                    }
                }
            }
        } else {
            document.cookie = "SEARCHKEYWORD" + "=" + keyword + "; path=/; expires=" + todayDate.toGMTString();
        }
    }

    function openHelpWinToPost(popupUrl, targetUrl, piWidth, piHeight, skin, popupId,left,top){

        if(left == "undefined" || left == ""){
            var xPos = (window.screen.width) ? (window.screen.width-piWidth)/2 : 0;
        }else{
            var xPos = left;
        }

        if(top == "undefined" || top == ""){
            var yPos = (window.screen.height) ? (window.screen.height-piHeight)/2 : 0;
        }else{
            var yPos = top;

        }

        var frm = $('<form></form>');
        var targetNm = popupId;

        if(popupId == ""){
            targetNm = 'HelpPopup';
        }

        frm.attr('action',popupUrl);
        frm.attr('target',targetNm);
        frm.attr('method','post');
        frm.append('<input type="hidden" name="Url" value="'+targetUrl+'">');
        frm.append('<input type="hidden" name="skin" value="'+skin+'">');
        $('body').append(frm);

        window.open('about:blank', targetNm, "top="+yPos+",left="+xPos+",width="+piWidth+",height="+piHeight+",status=no,toolbar=no,menubar=no,location=no,scrollbars=yes,resizable=yes");

        frm.submit();
    }

    var updateZoom = function(zoom){
        var currentSize = 0;
        if(zoom == 0){
            currentSize = 1.0;
        }else{
            currentSize += size+zoom*0.05;
        }

        this.size = currentSize;

        $('body').css('-webkit-transform','scale('+this.size+')');
        $('body').css('-moz-transform','scale('+this.size+')');
        $('body').css('-o-transform','scale('+this.size+')');

        $('body').css('-webkit-transform-origin','50% 0%');
        $('body').css('-moz-transform-origin','50% 0%');
        $('body').css('-o-transform-origin','50% 0%');

        $('body').css('transform','scale('+this.size+')');
        $('body').css('transform-origin','50% 0%');

    }
